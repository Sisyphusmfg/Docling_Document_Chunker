## 6. Usage Examples:

### Basic Usage:
from docling_processor import DoclingProcessor, ChunkConfig

# Create processor with custom chunking settings
config = ChunkConfig(max_chunk_size=1000, overlap_size=200)
processor = DoclingProcessor(config)

# Process a single file
parsed_doc = processor.parse_document("your_document.pdf")
chunks = processor.chunk_text(parsed_doc['content'], "your_document.pdf")

# Process entire directory (will skip .gdoc files)
all_chunks = processor.process_directory("./documents", ['.pdf', '.docx', '.doc'])

# Save chunks to JSON files
processor.save_chunks_to_json(all_chunks, r"D:\Docling Trial\Chunks")
